maithili = {
	trigger = {
		OR = {
			capital_county = {
				title:d_tirabhukti = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			has_primary_title = title:d_tirabhukti
		}
		culture = culture:bihari
		any_sub_realm_county = {
			OR = {
				culture = culture:bengali
				culture = culture:nepali
			}
			duchy = title:d_tirabhukti
		}
	}
}
