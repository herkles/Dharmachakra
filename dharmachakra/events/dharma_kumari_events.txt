dharma_nepal.100 = {# Establishing the Tradition
	type = character_event
	title = dharma_nepal.100.t
	desc = dharma_nepal.100.desc
	theme = faith
	left_portrait = {
		character = root
		animation = worry
	}
	
	immediate = {
		random_child = {
			limit = {
				age <= 8
				age >= 2
				culture = culture:nepali
				OR = {
					faith.religion = faith:mahayana.religion
					faith.religion = faith:vajrayana.religion
					religion = religion:hinduism_religion
					religion = religion:mundhum_religion
				}				
			}
			save_scope_as = kumari
		}
	}

	option = {# A new Nepali Tradition
		name = dharma_nepal.100
	}
}

dharma_nepal.101 = {# Kumari needs replacement
	type = character_event
	title = dharma_nepal.102.t
	desc = {
		triggered_desc = {# Kumari died of illness
			trigger = {
				scope:kumari = {
					death_reason = death_ill
				}
			}
			desc = religious_decision.102.desc.illness
		}
		triggered_desc = {# Kumari was killed
			trigger = {
				scope:kumari = {
					death_reason = death_murder
				}
			}
			desc = religious_decision.102.desc.murdered
		}
		triggered_desc = {# Kumari turned of age
			trigger = {
				scope:kumari = {
					death_reason = death_old_age
				}
			}
			desc = religious_decision.102.desc.woman
		}
	}
	theme = grief
	left_portrait = {
		character = root
		animation = worry
	}
	
	immediate = {

	}

	option = {# The gods have cursed us! - death by illness
		name = dharma_nepal.102.a.illness
		
	}
	
	option = {# a great sacriliage has taken place!
		name = dharma_nepal.102.a.murdered
	}

	option = {# The Kumari has become a woman
		name = dharma_nepal.102.a.woman
	}

	after = {
		clear_saved_scope = kumari
		trigger_event = { id = dharma_nepal.102 days = 30 }
	}
}
