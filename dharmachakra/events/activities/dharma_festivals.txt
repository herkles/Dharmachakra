dharma_festivals.106 = {# Giving the gift to your loved ones
	type = character_event
	title = dharma_festivals.106.t
	desc = {
		first_valid = {
			triggered_desc = {# Multiple Spouses
				trigger = {
					has_character_flag = diwali_wives_gift
					any_spouse = { count >= 2 }
				}
				desc = dharma_festivals.106.desc.spouses
			}
			triggered_desc = {# first spouse only
				trigger = {
					has_character_flag = diwali_first_spouse_gift
					any_spouse = { count >= 2 }
				}
				desc = dharma_festivals.106.desc.first_spouse
			}
			triggered_desc = {# My lover is not one of my wives
				trigger = {
					has_character_flag = diwali_lover_gift
					any_spouse = {
						NOT = { this = scope:lover }
					}				
				}
				desc = dharma_festivals.106.desc.lover_not_wife
			}
			triggered_desc = {# Lover is one of my wives but not first wife
				trigger = {
					has_character_flag = diwali_lover_gift
					any_spouse = { count >= 2 }
					primary_spouse = {
						NOT = { this = scope:lover }
					}		
				}
				desc = dharma_festivals.106.desc.lover_wife
			}
			desc = dharma_festivals.106.desc.spouse
		}
	}	
	theme = feast_activity
	override_background = {
		event_background = dharma_india_balcony
	}
	
	left_portrait = {
		character = root
		animation = happiness
	}

	immediate = {
		primary_spouse = {
			save_scope_as = first_spouse
		}
	}

	option = {# Your spouse gets a gift
		name = dharma_festivals.106.spouse
		add_piety = minor_piety_gain
		every_spouse = {
			limit = { is_alive = yes }
			add_gold = minor_gold_value
			add_opinion = {
				modifier = pleased_opinion
				target = root
				opinion = 15
			}
		}
	}
	option = {# Your wives get a gift
		name = dharma_festivals.106.spouses
		add_piety = minor_piety_gain
		every_spouse = {
			limit = { is_alive = yes }
			add_gold = minor_gold_value
			add_opinion = {
				modifier = pleased_opinion
				target = root
				opinion = 15
			}
		}
	}
	option = {# Your first wife gets a gift
		name = dharma_festivals.106.first_wife
		add_piety = minor_piety_gain
		primary_spouse = {
			add_gold = minor_gold_value
			add_opinion = {
				modifier = pleased_opinion
				target = root
				opinion = 15
			}
		}
		every_spouse = {
			limit = { 
				is_alive = yes 
				NOT = { primary_spouse = scope:first_spouse }
			}
			add_opinion = {# They don't like you for not getting a gift
				modifier = disappointed_opinion
				target = root
				opinion = -10
			}
			add_opinion = {# They are envious of your First Wife
				modifier = envy_opinion
				target = scope:first_spouse
				opinion = 15
			}
		}

	}
	option = {# Your lover gets a gift
		name = dharma_festivals.106.lover
		add_piety = minor_piety_gain
		scope:lover = {
			add_opinion = {
				target = root
				modifier = lover_breakup_opinion
				opinion = 15
			}
		}
		every_spouse = {
			limit = { NOT = { this = scope:lover } }
			add_opinion = {# They don't like you for not getting a gift
				target = root
				modifier = disappointed_opinion
				opinion = -10
			}
			add_opinion = {# They are envious of your lover
				target = scope:lover
				modifier = envy_opinion
				opinion = -15
			}
		}
	}

	after = {
		trigger_event = { 
			id = dharma_festivals.107
			days = 1
		}
	}

}

dharma_festivals.108 = {# pachikalu dice game
	type = character_event
	title = dharma_festivals.108.t
	desc = dharma_festivals.108.desc
	theme = friendly
	
	left_portrait = {
		character = root
		animation = happiness
	}

	option = {# play dice game
		name = dharma_festivals.108.play
		remove_short_term_gold = tiny_gold_value
		random_list = {
			50 = { add_gold = minor_gold_value	}
			50 = { }
		}
		stress_impact = {
			gregarious = miniscule_stress_impact_loss
			profligate = minor_stress_impact_loss

			shy = miniscule_stress_impact_gain
			diligent = miniscule_stress_impact_gain
		}
	}
	option = {# don't play
		name = dharma_festivals.108.dont_play
		stress_impact = {
			gregarious = miniscule_stress_impact_gain
			profligate = minor_stress_impact_gain

			shy = miniscule_stress_impact_loss
			diligent = miniscule_stress_impact_loss
		}
	}

	after = {
		if = {
			limit = {
				any_sibling = { count >= 2 }
				OR = {
					AND = {# Got to have a brother
						is_female = yes
						any_sibling = {
							NOT = { is_female = yes }
						}
					}
					AND = {# got to have a sister
						NOT = { is_female = yes }
						any_sibling = {
							is_female = yes
						}
					}
				}
			}
			trigger_event = { 
				id = dharma_festivals.109
				days = 1
			}
		}
		else = {
			trigger_event = { 
				id = dharma_festivals.110
				days = 1
			}
		}
	}
